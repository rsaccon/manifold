73inline HOST_DEVICE int CCW(glm::vec2 p0, glm::vec2 p1, glm::vec2 p2,
74 float tol) {
75 glm::vec2 v1 = p1 - p0;
76 glm::vec2 v2 = p2 - p0;
77 float area = v1.x * v2.y - v1.y * v2.x;
78 float base2 = glm::max(glm::dot(v1, v1), glm::dot(v2, v2));
79 if (area * area <= base2 * tol * tol)
80 return 0;
81 else
82 return area > 0 ? 1 : -1;
83}

113inline HOST_DEVICE float sind(float x) {
114 if (!std::isfinite(x)) return sin(x);
115 if (x < 0.0f) return -sind(-x);
116 int quo;
117 x = remquo(fabs(x), 90.0f, &quo);
118 switch (quo % 4) {
119 case 0:
120 return sin(glm::radians(x));
121 case 1:
122 return cos(glm::radians(x));
123 case 2:
124 return -sin(glm::radians(x));
125 case 3:
126 return -cos(glm::radians(x));
127 }
128 return 0.0f;
129}

144inline HOST_DEVICE glm::mat4x3 RotateUp(glm::vec3 up) {
145 up = glm::normalize(up);
146 glm::vec3 axis = glm::cross(up, {0, 0, 1});
147 float angle = glm::asin(glm::length(axis));
148 if (glm::dot(up, {0, 0, 1}) < 0) angle = glm::pi<float>() - angle;
149 return glm::mat4x3(glm::rotate(glm::mat4(1), angle, axis));
150}

277struct Box {
278 glm::vec3 min = glm::vec3(std::numeric_limits<float>::infinity());
279 glm::vec3 max = glm::vec3(-std::numeric_limits<float>::infinity());
280
284 HOST_DEVICE Box() {}
285
289 HOST_DEVICE Box(const glm::vec3 p1, const glm::vec3 p2) {
290 min = glm::min(p1, p2);
291 max = glm::max(p1, p2);
292 }
293
297 HOST_DEVICE glm::vec3 Size() const { return max - min; }
298
302 HOST_DEVICE glm::vec3 Center() const { return 0.5f * (max + min); }
303
308 HOST_DEVICE float Scale() const {
309 glm::vec3 absMax = glm::max(glm::abs(min), glm::abs(max));
310 return glm::max(absMax.x, glm::max(absMax.y, absMax.z));
311 }
312
316 HOST_DEVICE bool Contains(const Box& box) const {
317 return glm::all(glm::greaterThanEqual(box.min, min)) &&
318 glm::all(glm::greaterThanEqual(max, box.max));
319 }
320
324 HOST_DEVICE void Union(const glm::vec3 p) {
325 min = glm::min(min, p);
326 max = glm::max(max, p);
327 }
328
332 HOST_DEVICE Box Union(const Box& box) const {
333 Box out;
334 out.min = glm::min(min, box.min);
335 out.max = glm::max(max, box.max);
336 return out;
337 }
338
346 HOST_DEVICE Box Transform(const glm::mat4x3& transform) const {
347 Box out;
348 glm::vec3 minT = transform * glm::vec4(min, 1.0f);
349 glm::vec3 maxT = transform * glm::vec4(max, 1.0f);
350 out.min = glm::min(minT, maxT);
351 out.max = glm::max(minT, maxT);
352 return out;
353 }

396 HOST_DEVICE bool DoesOverlap(const Box& box) const {
397 return min.x <= box.max.x && min.y <= box.max.y && min.z <= box.max.z &&
398 max.x >= box.min.x && max.y >= box.min.y && max.z >= box.min.z;
399 }

405 HOST_DEVICE bool DoesOverlap(glm::vec3 p) const { // projected in z
406 return p.x <= max.x && p.x >= min.x && p.y <= max.y && p.y >= min.y;
407 }
